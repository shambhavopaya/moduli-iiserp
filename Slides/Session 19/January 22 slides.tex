\begin{frame}
	\begin{theorem}[Altman-Kleiman]
		Let $S$ be a Noetherian scheme and $V$ a vector bundle on $S$. Let $\pi: X\rightarrow S$ be a closed subscheme of $\P(V)$ and $L:=\sO_{\P(V)}|_X$. Let $W$ be a vector bundle on $S$ and $\nu$ an integer. Consider a coherent quotient $\pi^*(W)(\nu)\rightarrow E$. Then for any $\Phi\in \Q[\lambda]$, the quot functor $\mathfrak{Quot}^{\Phi}_{E/X/S}$ is representable by a scheme.
		
		Moreover, this scheme can be embedded in $\P(F)$ for some vector bundle $F$ over $S$.
	\end{theorem}
	\underline{Steps}:
	
	\begin{enumerate}
	\item Reduction to the case of $\mathfrak{Quot}^{\Phi}_{\pi*W/\P(V)/S}$
	
	\item Use of m-regularity
	
	\item Use of semi-continuity
	
	\item Functor to Grassmannian
	
	\item Use of flattening stratification
	\end{enumerate}
	
\end{frame}

\begin{frame}
		\begin{theorem}[Grothendieck]
		\label{theorem-quot-representable}
		Let $\pi: X\rightarrow S$ be a projective morphism with $S$ Noetherian. Then for any coherent sheaf $E$ on $X$ and any polynomial $\phi \in \Q[t]$, the functor $\mathfrak{Quot}^{\phi(t)}_{E/X/S}$ is representable by a projective $S$-scheme.
	\end{theorem}
\end{frame}
